\section*{Supplementary Figure S2: Exon-Embedded L1 and Baseline \msix{}--Poly(A)}

\begin{figure}[H]
\centering
\includegraphics[width=\textwidth]{figS3.pdf}
\caption{\textbf{Exon-embedded L1 are protected and baseline \msix{}--poly(A) is uncorrelated.}
\textbf{(a)}~Non-embedded L1 shows arsenite-induced shortening ($\Delta = -31$~nt), while embedded L1 is unaffected ($\Delta = -4$~nt, ns).
\textbf{(b)}~Median poly(A) by \msix{}/kb quartile under baseline conditions across nine cell lines. The four quartiles are indistinguishable ($r = -0.003$, ns), confirming that the \msix{}--poly(A) association emerges only under stress.}
\end{figure}

\section*{Supplementary Figure S3: Regulatory L1 Vulnerability}

\begin{figure}[H]
\centering
\includegraphics[width=\textwidth]{figS5.pdf}
\caption{\textbf{Regulatory L1 elements are selectively vulnerable to arsenite-induced poly(A) shortening.}
\textbf{(a)}~ECDF of poly(A) for regulatory vs.\ non-regulatory ancient L1. Regulatory L1 shows $\Delta = -73$~nt under stress, approximately three times the non-regulatory effect ($\Delta = -24$~nt).
\textbf{(b)}~Fraction of reads with poly(A) $<$30~nt by ChromHMM state.
\textbf{(c)}~\msix{}-matched comparison: regulatory L1 is 49--84~nt shorter than non-regulatory at equivalent \msix{} levels.
\textbf{(d)}~Within stressed regulatory L1, Q4 (highest \msix{}) shows median 59~nt vs.\ Q1 at 28~nt ($P = 2.9 \times 10^{-4}$).}
\end{figure}

\section*{Supplementary Figure S8: Sequence Features of Young L1 Confer Stress Immunity}

\begin{figure}[H]
\centering
\includegraphics[width=\textwidth]{fig4.pdf}
\caption{\textbf{Sequence features of young L1 confer stress immunity.}
\textbf{(a)}~Young vs.\ ancient L1 comparison across five immunity-associated features.
\textbf{(b)}~Stress-induced poly(A) change for ancient L1 stratified by presence/absence of each feature. Ancient elements possessing any single feature are immune to shortening.
\textbf{(c)}~Composite immunity score (0--3) predicts poly(A) retention dose-dependently (Spearman $\rho = 0.169$, $P = 5.5 \times 10^{-18}$).
\textbf{(d)}~DRACH and CpG density across L1 functional domains. The EN domain combines the highest DRACH density with near-complete CpG depletion.}
\end{figure}
